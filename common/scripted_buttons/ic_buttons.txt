sb_international_crisis_territory_request_help_oppo = {
	desc = sb_international_crisis_territory_request_help_desc
	name = sb_international_crisis_territory_request_help_oppo
	visible = {
		this = global_var:terri_crisis_opposite
	}
	possible = {
		this = global_var:terri_crisis_opposite
	}
	effect = {
		every_country = {
			limit = {
				NOR = {
					this = global_var:terri_crisis_opposite
					this = global_var:terri_crisis_target
				}
				is_in_any_com_situation_faction = yes
				#반대편에 있으면 못 참가하게 하기
			}
			trigger_event = ic_territory.4
		}
		
	}
}
