    on_monthly_pulse = {
        random_events = {
            200 = 0
            #10 = ic_territory.10
        }
    }
    
	immediate = {
        every_country = {
            limit = {
                this = global_var:terri_crisis_opposite
            }
            save_scope_as = terri_crisis_opposite
        }
        every_country = {
            limit = {
                this = global_var:terri_crisis_target
            }
            save_scope_as = terri_crisis_target
        }

            
        
        create_com_situation = {
            situation_id = je_com_international_crisis_territory

            side_name_left = je_com_international_crisis_territory_left
            side_desc_left = je_com_international_crisis_territory_left_desc

            side_leader_country_right = scope:terri_crisis_target
            side_name_right = je_com_international_crisis_territory_right
            side_desc_right = je_com_international_crisis_territory_right_desc
        }

            text = je_com_international_crisis_territory_header

        create_com_situation_faction = {
            side = left
            name = je_com_international_crisis_territory_left
            desc = je_com_international_crisis_territory_desc
        }
        add_com_situation_faction_country = {
            side = left
            name = je_com_international_crisis_territory_left
            country = scope:terri_crisis_opposite
        }


        # Create tar
        create_com_situation_faction = {
            side = right
            name = je_com_international_crisis_territory_right
            desc = je_com_international_crisis_territory_right_desc
        }
        add_com_situation_faction_country = {
            side = right
            name = je_com_international_crisis_territory_right
            country = scope:terri_crisis_target
        }
		
        # Create Bystanders faction
        create_com_situation_faction = {
            side = neutral
            name = je_com_international_crisis_territory_faction_bystanders
            desc = je_com_international_crisis_territory_faction_bystanders_desc
        }
        add_com_situation_faction_country = {
            side = neutral
            name = je_com_international_crisis_territory_faction_bystanders
        }

		
        # Create Main Actions button group
        create_com_situation_button_group = {
            name = je_com_international_crisis_territory_button_group_main_actions
            desc = je_com_international_crisis_territory_button_group_main_actions_desc
        }
        add_com_situation_button_group_element = {
            group = je_com_international_crisis_territory_button_group_main_actions
            name = none

        create_com_situation_button_group = {

        add_com_situation_button_group_element = {
            group = je_com_international_crisis_territory_button_group_diplomatic_actions
            name = sb_international_crisis_territory_request_help_oppo
        }
        add_com_situation_button_group_element = {
            group = je_com_international_crisis_territory_button_group_diplomatic_actions
            name = sb_international_crisis_territory_request_help_tar
        }
        ordered_country = {
            limit = {
                and = {

                    NOT = {
                        AND = {
